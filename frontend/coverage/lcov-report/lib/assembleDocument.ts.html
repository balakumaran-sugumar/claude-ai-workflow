export function getMndaTermText(values: NdaFormValues): string {
  return values.mndaTerm.type === 'fixed'
    ? `${values.mndaTerm.years} year(s) from Effective Date`
    : 'termination by either party in writing';
}
 
export function getConfidentialityText(values: NdaFormValues): string {
  return values.termOfConfidentiality.type === 'fixed'
    ? `${values.termOfConfidentiality.years} year(s) from Effective Date`
    : 'in perpetuity';
}
 
export function getFieldValues(values: NdaFormValues): Record<string, string> {
  return {
    Purpose: values.purpose,
    'Effective Date': values.effectiveDate,
    'MNDA Term': getMndaTermText(values),
    'Term of Confidentiality': getConfidentialityText(values),
    'Governing Law': values.governingLaw,
    Jurisdiction: values.jurisdiction,
  };
}
 
export function assembleCoverPage(values: NdaFormValues): string {
  const mndaTermLines =
    values.mndaTerm.type === 'fixed'
      ? `- [x] Expires ${values.mndaTerm.years} year(s) from Effective Date.\n- [ ] Continues until terminated in accordance with the terms of the MNDA.`
      : `- [ ] Expires [N] year(s) from Effective Date.\n- [x] Continues until terminated in accordance with the terms of the MNDA.`;
 
  const confidentialityLines =
    values.termOfConfidentiality.type === 'fixed'
      ? `- [x] ${values.termOfConfidentiality.years} year(s) from Effective Date, but in the case of trade secrets until Confidential Information is no longer considered a trade secret under applicable laws.\n- [ ] In perpetuity.`
      : `- [ ] [N] year(s) from Effective Date, but in the case of trade secrets until Confidential Information is no longer considered a trade secret under applicable laws.\n- [x] In perpetuity.`;
 
  const modificationsText = values.modifications.trim() || 'None';
 
  return `# Mutual Non-Disclosure Agreement
 
## USING THIS MUTUAL NON-DISCLOSURE AGREEMENT
 
This Mutual Non-Disclosure Agreement (the "MNDA") consists of: (1) this Cover Page ("**Cover Page**") and (2) the Common Paper Mutual NDA Standard Terms Version 1.0 ("**Standard Terms**") identical to those posted at [commonpaper.com/standards/mutual-nda/1.0](https://commonpaper.com/standards/mutual-nda/1.0). export function assembleStandardTerms(values: NdaFormValues, forPreview: boolean): string {
  const fieldValues = getFieldValues(values);
  let result = STANDARD_TERMS_RAW;
 
  for (const [field, value] of Object.entries(fieldValues)) {
    const escaped = field.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    const regex = new RegExp(`<span class="coverpage_link">${escaped}</span>`, 'g');
    const replacement = forPreview ? `<mark class="nda-chip">${value}</mark>` : value;
    result = result.replace(regex, replacement);
  }
 
  return result;
}
 
export function assembleFullDocument(values: NdaFormValues): string {
  const coverPage = assembleCoverPage(values);
  const standardTerms = assembleStandardTerms(values, false);
  return `${coverPage}\n\n---\n\n${standardTerms}`;
}
